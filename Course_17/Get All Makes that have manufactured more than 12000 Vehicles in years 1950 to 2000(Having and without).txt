


	select Make, NumbersOfVehicle = count(Makes.MakeID) from Makes
	inner join VehicleDetails on VehicleDetails.MakeID = Makes.MakeID where year between 1950 and 2000

	group by Make
	having COUNT(Makes.MakeID) > 12000
	order by NumbersOfVehicle desc
