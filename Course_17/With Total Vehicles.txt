

select Makes.Make, NumbersOfVehicle = count(Makes.MakeID), TotalVeichles = (select top 1 count(MakeID) from VehicleDetails) from Makes
inner join VehicleDetails on VehicleDetails.MakeID = Makes.MakeID
where (year between 1950 and 2000)
group by Makes.Make
order by NumbersOfVehicle desc;
