#include <iostream>
#include <string>

using namespace std;

string ReadString(string message) 
{
	string text;

	cout << message << endl;
	getline(cin, text);

	return text;
}

char InvertChar(char letter) 
{
	return isupper(letter) ? tolower(letter) : toupper(letter);
}

string InvertString(string text) 
{
	for (short i = 0; i < text.length(); i++) 
	{
		text[i] = InvertChar(text[i]);
	}

	return text;
}

int main()
{
	string text = ReadString("Plz enter your string : \n");

	cout << endl << "Invert string is : " << endl;
	cout << InvertString(text) << endl;


	system("pause > 0");

	return 0;
}
