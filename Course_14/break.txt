using System;

namespace ConsoleApp2
{
    internal class Program
    {
        static void Main(string[] args)
        {


            int i = 2;

            do
            {

                Console.WriteLine("C# while Loop: Iteration {0}", i);

                if (i == 6)
                    break;

                i++;

            } while (i <= 4);

            Console.ReadKey();

        }
    }
}
