using System;
using System.Diagnostics;

class Program
{
    static void Main()
    {
        Console.WriteLine("Running processes:\n");

        Process[] processes = Process.GetProcesses();

        int count = 0;

        foreach (Process process in processes)
        {
            try
            {
                Console.WriteLine($"Name: {process.ProcessName}, ID: {process.Id}");
                count++;
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error reading process: {ex.Message}");
            }
        }

        Console.WriteLine($"\nTotal number of processes: {count}");

        Console.WriteLine("\nPress any key to exit...");
        Console.ReadKey();
    }
}
