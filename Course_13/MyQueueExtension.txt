#pragma once

#include<iostream>
#include "clsDblLinkedList.h"

using namespace std;

template <typename t> class clsQueue
{

protected:
	clsDblLinkedList<t> DblLinkedList;

public:
	
	void push(t value) 
	{
		DblLinkedList.InsertAtEnd(value);
	}

	void pop() 
	{
		DblLinkedList.DeleteFirstNode(DblLinkedList.head);
	}

	int Size() 
	{
		return DblLinkedList.Size();
	}

	t front() 
	{	
		return DblLinkedList.GetItem(0);
	}

	t back() 
	{
		return DblLinkedList.GetItem(Size() - 1);
	}

	bool IsEmpty() 
	{
		return DblLinkedList.IsEmpty();
	}

	void Print() 
	{
		DblLinkedList.PrintList();
	}

	t GetItem(int index) 
	{
		return DblLinkedList.GetItem(index);
	}

	void Reverse() 
	{
		DblLinkedList.Reverse();
	}

	bool UpdateItem(int index, t newValue) 
	{
		return DblLinkedList.UpdateItem(index, newValue);
	}

	void InsertAfter(int index, t newValue) 
	{
		DblLinkedList.InsertAfter(index, newValue);
	}

	void InsertAtFront(t newValue) 
	{
		DblLinkedList.InsertAtBeginning(newValue);
	}

	void InsertAtBack(t newValue)
	{
		DblLinkedList.InsertAtEnd(newValue);
	}

	void Clear()
	{
		DblLinkedList.Clear();
	}

};
