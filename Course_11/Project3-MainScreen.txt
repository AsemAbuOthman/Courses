#pragma once

#include<iostream>
#include<string>
#include<iomanip>
#include "clsInputValidate.h"
#include "clsScreen.h"

using namespace std;

class clsMainScreen : protected clsScreen
{
private:

	enum enMainMenuOptions {eListClients = 1, eAddNewClient = 2, eDeleteClient = 3, eUpdateClient = 4, eFindClient = 5, eShowTransactionsMenu = 6, eManageUsers = 7, eExit = 8};

	static short _ReadMainMenuOption() 
	{
		short option = 0;

		cout << setw(37) << left << "" << "Choose what do you want to do : ";
		option = clsInputValidate::ReadIntNumberBetween(1, 8, "Enter number between 1 to 8 ? ");

		return option;
	}

	static void _ShowAllClientList()
	{
		cout << "Here will be Show All Client List Screen.." << endl;
	
	};

	static void _GoBackToMainMenu()
	{
		cout << "Here will be Show Main Menu Screen.." << endl;

	};

	static void _ShowAddClientsScreen()
	{
		cout << "Here will be Show Main Menu Screen.." << endl;

	}

	static void _ShowDeleteClientsScreen()
	{
		cout << "Here will be Show Delete Screen.." << endl;

	}

	static void _ShowUpdateClientsScreen()
	{
		cout << "Here will be Show Update Screen.." << endl;

	}

	static void _ShowFindClientsScreen()
	{
		cout << "Here will be Show Find Screen.." << endl;

	}

	static void _ShowTransactionsMenu()
	{
		cout << "Here will be Show Transaction Menu Screen.." << endl;

	}

	static void _ShowManageUsers()
	{
		cout << "Here will be Show Manage Users Screen.." << endl;

	}

	static void _ShowEndScreen()
	{
		cout << "Here will be Show End Screen.." << endl;

	}

	static void _PerformMainMenuOption(enMainMenuOptions MainMenuOption)
	{
		switch (MainMenuOption)
		{
		case clsMainScreen::eListClients:

			system("cls");
			_ShowAllClientList();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eAddNewClient:

			system("cls");
			_ShowAddClientsScreen();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eDeleteClient:

			system("cls");
			_ShowDeleteClientsScreen();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eUpdateClient:

			system("cls");
			_ShowUpdateClientsScreen();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eFindClient:

			system("cls");
			_ShowFindClientsScreen();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eShowTransactionsMenu:

			system("cls");
			_ShowTransactionsMenu();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eManageUsers:
			
			system("cls");
			_ShowManageUsers();
			_GoBackToMainMenu();

			break;
		case clsMainScreen::eExit:

			system("cls");
			_ShowEndScreen();

			break;

		}
	}

public:

	static void ShowMainMenu() 
	{
		system("cls");

		_DrawScreenHeader("\t\tMain Menu");

		cout << setw(37) << left << "" << "=====================================" << endl;
		cout << setw(37) << left << "" << "\t\tMain Menu\n" << endl;
		cout << setw(37) << left << "" << "=====================================" << endl;

		cout << setw(37) << left << "" << "\t[1] Show Client List. " << endl;
		cout << setw(37) << left << "" << "\t[2] Add New Client. " << endl;
		cout << setw(37) << left << "" << "\t[3] Delete Client. " << endl;
		cout << setw(37) << left << "" << "\t[4] Update Client. " << endl;
		cout << setw(37) << left << "" << "\t[5] Find Client. " << endl;
		cout << setw(37) << left << "" << "\t[6] Transactions. " << endl;
		cout << setw(37) << left << "" << "\t[7] Manage Users. " << endl;
		cout << setw(37) << left << "" << "\t[8] Exit. " << endl;
		cout << setw(37) << left << "" << "=====================================" << endl;


		_PerformMainMenuOption((enMainMenuOptions)_ReadMainMenuOption());
 	}

};
