namespace OOP_C_
{
    public class clsPerson 
    {
        public string Name { get; set; }
        public int Age { get; set; }

        public void Introduce() 
        {
            Console.WriteLine($"Hi, I'm {Name} and my age is {Age}");
        }

    }

    public class clsEmployee : clsPerson
    {
        public int EmployeeId { get; set; }
        public float Salary { get; set; }

        public void Work() 
        {
            Console.WriteLine($"Employee wirh ID {EmployeeId} and Salary {Salary:C} is working");
        }
    
    }

    public class clsDoctor : clsEmployee 
    {
    
        public string Spicality { get; set; }

        public void Heal() 
        {
            Console.WriteLine($"Doctor {Name} with ID {EmployeeId}, Salary {Salary:C}, and Spicality {Spicality}");
        }
    }

    internal class Program
    {
        static void Main(string[] args)
        {
         
            clsDoctor doctor = new clsDoctor();

            doctor.Name = "Asem";
            doctor.Age = 22;
            doctor.EmployeeId = 3200100;
            doctor.Salary = 5000;
            doctor.Spicality = "Denteist";
            doctor.Introduce();
            doctor.Work();
            doctor.Heal();

            Console.ReadKey();
        }
    }
}
